def scale_0_100(df):
    return (df - df.min()) / (df.max() - df.min()) * 100

mean_summary = df.groupby('KmeansCluster')[summary_features].mean()
median_summary = df.groupby('KmeansCluster')[summary_features].median()
mean_summary_scaled = scale_0_100(mean_summary)
median_summary_scaled = scale_0_100(median_summary)
print(mean_summary_scaled)
print(median_summary_scaled)


# Sort mean summary by cluster ascending
mean_summary_sorted = mean_summary_scaled.sort_index(ascending=True)

# Sort median summary by cluster ascending
median_summary_sorted = median_summary_scaled.sort_index(ascending=True)

print("=== Mean Summary by Cluster (Sorted Ascending) ===")
print(mean_summary_sorted)

print("\n=== Median Summary by Cluster (Sorted Ascending) ===")
print(median_summary_sorted)
